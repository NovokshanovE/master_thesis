\begin{frame}[shrink=5]\frametitle{Анализ существующих подходов}
\begin{itemize}\setlength{\itemsep}{0.2em}
  \item \textbf{Внешние справочники «IP-префикс → AS»:} просты во внедрении, обновление раз в сутки, требуется ручная загрузка; непригодны как единственный источник для систем реального времени.
  \item \textbf{Ranger (существующее хранилище):} поддерживает только полный снимок; одиночная вставка требует перестройки всей таблицы --- $O(n)$ (1\,М$+$ маршрутов IPv4, 250\,тыс.\ IPv6).
\end{itemize}
\vspace{0.4em}
\begin{block}{Требования к новому решению}
\begin{itemize}\setlength{\itemsep}{0.15em}
  \item Поддержка потока обновлений --- снижение задержки обновления таблицы.
  \item Одиночная вставка/обновление без перестройки всей таблицы.
  \item Сохранение производительности по существующим операциям.
\end{itemize}
\end{block}
\end{frame}

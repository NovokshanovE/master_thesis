% ==================== Section: Context ====================
\section{Контекст и актуальность}

\begin{frame}{Обзор предметной области}
\begin{itemize}
  \item \textbf{AS (Autonomous System)} — административная единица маршрутизации Интернета
  \item \textbf{BGP (Border Gateway Protocol)} — протокол обмена маршрутной информацией между AS
  \item Целевое представление данных: \texttt{prefix → ASN}
\end{itemize}

\vspace{0.7em}
\alert{\faNetworkWired\quad AS path (путь автономных систем) позволяет построить граф связности Интернета. Анализируя этот граф, можно определить, какой автономной системе (организации) принадлежит конкретный IP-адрес или префикс.}
\end{frame}

\begin{frame}{Где это используется}
\begin{columns}[T,onlytextwidth]
\column{0.62\textwidth}
\begin{itemize}
  \item DDoS mitigation (фильтрация / политика обработки трафика)
  \item Анализ и классификация сетевого трафика
  \item Продукты: \textbf{BIFIT Mitigator} и \textbf{BIFIT Collector}
\end{itemize}

\column{0.38\textwidth}
\vspace{0.4em}
\begin{block}{Ключевая потребность}
\small
Быстро и точно отвечать на вопрос: \emph{«К какой AS относится IP/префикс?»}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Актуальность}
\begin{itemize}
  \item Распространённый подход: GeoIP/ASN базы (обновление раз в часы/дни)
  \item BGP-роутинг изменяется динамически (слияния сетей, анонсы новых префиксов, сбои)
  \item Устаревшие данные → неверные решения в защите/аналитике
\end{itemize}

\vspace{0.7em}
\alert{\faClock\quad Требуется актуализация AS-базы в режиме реального времени.}
\end{frame}

\begin{frame}{Структура данных: Adaptive Radix Tree (ART)}
\begin{columns}[T,onlytextwidth]
\column{0.60\textwidth}
\begin{itemize}
  \item Adaptive Radix Tree (ART) — разновидность префиксного дерева (trie) с узлами переменного размера.
  \item Узлы меняют формат (4/16/48/256 детей) \textbf{в зависимости от ветвления}
  \item Быстрый поиск по байтам ключа (IPv4/IPv6 префиксы)
  \item Хороша для:
  \begin{itemize}
    \item потоковой выгрузки
    \item компактного хранения ключей
  \end{itemize}
\end{itemize}

\vspace{0.4em}
\alert{\faLightbulb\quad В каждом шарде: \texttt{map} для O(1) Get/Upsert и \texttt{ART} для быстрого обхода.}

\column{0.3\textwidth}
\begin{block}{Почему не только map?}
\small
\begin{itemize}
  \item Хэш-таблица (map) не поддерживает упорядоченный обход, необходимый для эффективного полного дампа (DumpAll).
  \item Полный дамп требует эффективной итерации без глобальной блокировки
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{ART: Структура внутренних узлов}
\begin{itemize}
  \item Ключ хранится как последовательность байтов (например, \texttt{IPv6} — 16 байт)
  \item Узлы: \textbf{Node4 → Node16 → Node48 → Node256} (по мере роста числа детей)
  \item Такой подход снижает накладные расходы на хранение узлов и ускоряет поиск в дереве за счёт адаптации к плотности ключей
\end{itemize}

\begin{frame}{Формулировка проблемы}
\begin{block}{}
\large
Отсутствует механизм \textbf{непрерывной актуализации} AS-базы в реальном времени на основе обновления маршрутной информации,
пригодный для \textbf{высоконагруженных систем}.
\end{block}

\vspace{0.7em}
\alert{\faExclamationTriangle\quad Нужен выделенный сервис, обновляющий таблицу \texttt{prefix → ASN} по мере обновления маршрутной информации.}
\end{frame}

% ==================== Section: Goals ====================
\section{Цель, задачи и план}

\begin{frame}{Цель и задачи}
\textbf{Цель:} разработать микросервис, поддерживающий AS-базу на основе BGP-данных в режиме реального времени.

\vspace{0.8em}
\textbf{Задачи:}
\begin{itemize}
  \item Спроектировать архитектуру микросервиса и контуры интеграции
  \item Реализовать взаимодействие с GoBGP (FullView + Update Stream)
  \item Реализовать gRPC API для потребителей (Collector / Mitigator)
  \item Подготовить методику и план экспериментов (latency, нагрузка, точность)
\end{itemize}
\end{frame}

\section{Текущий прогресс}

\begin{frame}{Что уже реализовано}
\begin{itemize}
  \item Выделена BGP-логика из Collector в отдельный микросервис
  \item Реализован клиент взаимодействия с GoBGP 
  \item Реализовано gRPC API для:
  \begin{itemize}
    \item общения Collector ↔ микросервис
    \item общения Mitigator ↔ микросервис
  \end{itemize}
\end{itemize}

\vspace{0.7em}
\alert{\faCheckCircle\quad Завершён первый этап: вся логика работы с BGP выделена из Collector в самостоятельный работоспособный микросервис с протестированным gRPC API}
\end{frame}

\begin{frame}{План работ на следующий семестр}
\begin{enumerate}
  \item Доработка и оптимизация хранилища \texttt{prefix → ASN}
  \item Внедрение и оптимизация алгоритма поиска ASN по IP-адресу (Longest Prefix Match)
  \item Потоковое применение BGP-обновлений (низкая задержка)
  \item Нагрузочное тестирование и бенчмарки
  \item Сравнение с подходом периодической FullView-обработки
\end{enumerate}
\end{frame}

\begin{frame}{Ожидаемые результаты}
\begin{itemize}
  \item Снижение задержки между BGP-изменением и фиксацией в AS-базе
  \item Улучшение качества решений Mitigator/Collector за счёт актуальности данных
  \item Масштабируемая архитектура, готовая к эксплуатации
\end{itemize}

\vspace{0.7em}
\alert{\faRocket\quad Практическая ценность: более точная и быстрая реакция на изменения в сети.}
\end{frame}
